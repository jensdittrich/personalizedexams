\definecolor{framecolor}{rgb}{0.6,0.6,0.6}

\newenvironment{solution}[1]
    {%
        \ifthenelse{\isundefined{\displaysolutions}}%
            {\expandafter\comment}%
            {\def\FrameCommand{\fcolorbox{framecolor}{shadecolor}}\MakeFramed {\FrameRestore} {\large \textbf{Solution:} {\normalsize #1}\\}}%
    }%
    {%
        \ifthenelse{\isundefined{\displaysolutions}}%
            {\expandafter\endcomment}%
            {\endMakeFramed}
    }
